Задачи раздела Begin

Приведённая выше информация позволит вам решить задачи из этой секции. Если же вы испытываете трудности, вы можете посмотреть, как программа должна отработать, а также посмотреть её полное решение.

Begin 1

Дана сторона квадрата a. Найти его периметр P = 4⋅ a. 

let a = Number(prompt("Введите сторону квадрата"));
let P = 4 * a;
alert("Периметр квадрата: " + P);
package java_tasks;

import java.util.Scanner;

class JavaTasks_Begin1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Введите сторону квадрата: ");
        float a = scanner.nextFloat();

        float P = 4 * a;

        System.out.println("Периметр квадрата: " + P);
        scanner.close();
    }
}
